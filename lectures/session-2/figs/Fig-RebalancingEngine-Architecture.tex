\def\R{72mm}

% --- Cardinal-position OODA stages ---
\node[observe] (observe) at (90:\R) {%
    {\bfseries\sffamily\LARGE\textcolor{rule}{1.\ OBSERVE}}\\[2pt]
    {\normalsize\itshape read the new bar}\\[5pt]
    \textbullet\ Market prices $\{S_t\}$\\[1pt]
    \textbullet\ SIM regression $\to (\alpha_i, \beta_i, \sigma_{\varepsilon,i})$\\[1pt]
    \textbullet\ EMA crossover $\to \lambda_t$
};

\node[orient] (orient) at (0:\R) {%
    {\bfseries\sffamily\LARGE\textcolor{cornellred}{2.\ ORIENT}}\quad\textcolor{cornellred}{\small\textbf{[AI step]}}\\[2pt]
    {\normalsize\itshape map features to preferences}\\[5pt]
    $\displaystyle \gamma_i = \tanh\!\Big(\tfrac{\alpha_i}{\beta_i^{\lambda_t}} + \beta_i^{1-\lambda_t}\,\mathbb{E}[g_{\mathrm{mkt}}]\Big)$\\[3pt]
    \textbullet\ Per-asset preference\\
    \phantom{\textbullet\ }$\gamma_i \in (-1, 1)$
};

\node[decide] (decide) at (-90:\R) {%
    {\bfseries\sffamily\LARGE\textcolor{calblue}{3.\ DECIDE}}\\[2pt]
    {\normalsize\itshape allocate, then check guardrails}\\[5pt]
    $\displaystyle n_i^{\star} = \tfrac{\gamma_i}{\sum_{j\in\mathcal{A}^+}\gamma_j} \cdot \tfrac{B_{\mathrm{adj}}}{P_i}$\\[3pt]
    \textbullet\ Schedule $b_t \in \{0, 1\}$\\[1pt]
    \textbullet\ Drawdown $d_{\max}$, turnover $\tau_{\max}$
};

\node[act] (act) at (180:\R) {%
    {\bfseries\sffamily\LARGE\textcolor{darkgreen}{4.\ ACT}}\\[2pt]
    {\normalsize\itshape execute trades, update state}\\[5pt]
    \textbullet\ Trade $\Delta n_i$ or hold\\[1pt]
    \textbullet\ Realize transaction costs\\[1pt]
    \textbullet\ Update $W_t = \sum_i n_i\, S_i(t)$
};
